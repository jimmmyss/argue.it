# Push Commands for jimmmyss/argue.it

## Step 1: Configure Git (run once)
git config --global user.name "jimmmyss"
git config --global user.email "your.email@gmail.com"

## Step 2: Initialize and Push Repository
git init
git add .
git commit -m "Initial commit: Secure ArgueIt platform"
git remote add origin https://github.com/jimmmyss/argue.it.git
git push -u origin main

## If the push fails, try:
git branch -M main
git push -u origin main

## To verify your configuration:
git config --list
